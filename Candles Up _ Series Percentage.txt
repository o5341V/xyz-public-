study("Candles Up / Series Percentage")

seriesLength = input(50)
upVar = 0
upVar := close > open ? 1 : 0 

hl2Var = 0
hl2Var := hl2 > hl2[1] ? 1 : 0

plot(sum(upVar,seriesLength)/seriesLength, color=color.green, title="close > open")
plot(sum(hl2Var,seriesLength)/seriesLength, color=color.aqua, title="hl2 > hl2[1]")
